\section {Verb Tenses}
\begin{itemize}

\item \textbf{Present Simple} is used for habitual actions, routines, general facts, and permanent situations.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + Verb (base form, add \textit{-s} for he, she, it)
	\item Negative: Subject + \textit{do not / does not} + Verb (base form)
	\item Question: \textit{Do / Does} + Subject + Verb (base form)
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{She plays soccer on weekends.}
	\item Negative: \textit{She does not play soccer on weekends.}
	\item Question: \textit{Does she play soccer on weekends?}
\end{itemize}

\item  \textbf{Past Simple} is used to describe actions or events that happened at a specific point in the past and are now finished.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + Verb in past (\textit{-ed} for regular verbs, specific form for irregular verbs)
	\item Negative: Subject + \textit{did not} + Verb (base form)
	\item Question: \textit{Did} + Subject + Verb (base form)
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{They visited Paris last year.}
	\item Negative: \textit{They did not visit Paris last year.}
	\item Question: \textit{Did they visit Paris last year?}
\end{itemize}

\item  \textbf{Present Perfect} is used to express past experiences without specifying when they occurred, actions that started in the past and continue in the present, or recent actions with relevance to the present.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + \textit{have / has} + Past Participle
	\item Negative: Subject + \textit{have not / has not} + Past Participle
	\item Question: \textit{Have / Has} + Subject + Past Participle
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{I have finished my homework.}
	\item Negative: \textit{I have not finished my homework.}
	\item Question: \textit{Have you finished your homework?}
\end{itemize}

\item  \textbf{Present Continuous} is used to describe actions happening at the moment of speaking or temporary situations.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + \textit{am / is / are} + Verb-ing
	\item Negative: Subject + \textit{am not / is not / are not} + Verb-ing
	\item Question: \textit{Am / Is / Are} + Subject + Verb-ing
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{She is studying English.}
	\item Negative: \textit{She is not studying English.}
	\item Question: \textit{Is she studying English?}
\end{itemize}

\item  \textbf{Past Continuous} is used to describe actions that were in progress at a specific time in the past.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + \textit{was / were} + Verb-ing
	\item Negative: Subject + \textit{was not / were not} + Verb-ing
	\item Question: \textit{Was / Were} + Subject + Verb-ing
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{They were playing soccer.}
	\item Negative: \textit{They were not playing soccer.}
	\item Question: \textit{Were they playing soccer?}
\end{itemize}

\item  \textbf{Future Simple} is used to talk about actions or events that will happen in the future.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + \textit{will} + Verb (base form)
	\item Negative: Subject + \textit{will not} + Verb (base form)
	\item Question: \textit{Will} + Subject + Verb (base form)
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{They will travel tomorrow.}
	\item Negative: \textit{They will not travel tomorrow.}
	\item Question: \textit{Will they travel tomorrow?}
\end{itemize}

\item  \textbf{Future Continuous} describes actions that will be in progress at a specific time in the future.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + \textit{will be} + Verb-ing
	\item Negative: Subject + \textit{will not be} + Verb-ing
	\item Question: \textit{Will} + Subject + \textit{be} + Verb-ing
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{She will be working at 5 PM.}
	\item Negative: \textit{She will not be working at 5 PM.}
	\item Question: \textit{Will she be working at 5 PM?}
\end{itemize}

\item  \textbf{Past Perfect} is used to express an action that happened before another action in the past.

\subitem {Structure}
\begin{itemize}
	\item Affirmative: Subject + \textit{had} + Past Participle
	\item Negative: Subject + \textit{had not} + Past Participle
	\item Question: \textit{Had} + Subject + Past Participle
\end{itemize}

\subitem {Examples}
\begin{itemize}
	\item Affirmative: \textit{They had left before I arrived.}
	\item Negative: \textit{They had not left before I arrived.}
	\item Question: \textit{Had they left before I arrived?}
\end{itemize}
\end{itemize}
